   ```c
       /**
         * 哈哈，Zset
         * 可以看出，zset由哈希表dict和跳表zskiplist实现
         * 这两种数据结构是如何使用的呢?
         * >> 解密: dict zsl均会保存一份该集合中的数据，dict的功能是为了快速检索，如获取元素分数指令:zscore
        */
       typedef struct zset {
    
          // 哈希表，为了快速检索(已经明确元素的检索,如zscore)
          dict *dict;

          // skiplist
          zskiplist *zsl;
       } zset;
   ```
